**Send a direct message**&#x20;

```

async function sendIndividualMessage(accountId, messageData) {
  const url = `https://api.a1base.com/v1/messages/individual/${accountId}/send`;

  const headers = {
    'Content-Type': 'application/json',
    'X-API-Key': 'YOUR_API_KEY',
    'X-API-Secret': 'SECRET'
  };

  const body = JSON.stringify({
    content: messageData.content,
    attachment_uri: messageData.attachment_uri,
    from: messageData.from,
    to: messageData.to,
    service: messageData.service,
  });

  try {
    const response = await fetch(url, {
      method: 'POST',
      headers: headers,
      body: body
    });

    if (!response.ok) {
      throw new Error(`Error: ${response.status} ${response.statusText}`);
    }

    const data = await response.json();
    console.log('Individual Message Sent:', data);
    return data;
  } catch (error) {
    console.error('Failed to send individual message:', error);
    throw error;
  }
}

// Usage example
const accountId = 'your_account_id_here';
const messageData = {
  content: "hello",
  attachment_uri: "",
  from: "61421868490", //Your agent's number
  to: "610000000",
  service: "whatsapp"
};

sendIndividualMessage(accountId, messageData);
 
```
